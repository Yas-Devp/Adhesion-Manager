<?php
    function checherDB($type, $donnee){
        global $conn;

        if ($type === "r_s") {
            $sql = "SELECT * FROM enterprise WHERE raison_social LIKE ?";
            $types_param = "s";
            $params = ["%$donnee%"];
        } elseif ($type === "ice") {
            $sql = "SELECT * FROM enterprise WHERE ice = ?";
            $types_param = "s";
            $params = [$donnee];
        } elseif ($type === "n") {
            $sql = "SELECT * FROM enterprise WHERE nom LIKE ?";
            $types_param = "s";
            $params = ["%$donnee%"];
        } elseif ($type === "p") {
            $sql = "SELECT * FROM enterprise WHERE prenom LIKE ?";
            $types_param = "s";
            $params = ["%$donnee%"];
        } elseif ($type === "nom_et_prenom") {
            // L'astuce magique : recherche le mot dans "prenom nom" OU "nom prenom"
            $sql = "SELECT * FROM enterprise WHERE CONCAT(prenom, ' ', nom) LIKE ? OR CONCAT(nom, ' ', prenom) LIKE ?";
            $types_param = "ss";
            $params = ["%$donnee%", "%$donnee%"];
        }

        $stmt = $conn->prepare($sql);
        if (!$stmt) {
            die("Preparation de requet a echoue : " . $conn->error);
        }

        // Liaison dynamique des paramètres pour gérer le cas "ss" ou "s"
        $stmt->bind_param($types_param, ...$params);
        
        $stmt->execute();
        $result = $stmt->get_result();
        $data = [];

        if ($result && $result->num_rows > 0) {
            while ($row = $result->fetch_assoc()) {
                $data[] = $row;
            }
        }

        $stmt->close();
        return $data;
    }
?>
